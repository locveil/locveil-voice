# API CREDENTIALS - Cloud Service Authentication
# ============================================================

# OpenAI API (for GPT models and Whisper)
# Get your API key from: https://platform.openai.com/api-keys
OPENAI_API_KEY=your_openai_api_key_here

# Anthropic API (for Claude models)
# Get your API key from: https://console.anthropic.com/
ANTHROPIC_API_KEY=your_anthropic_api_key_here

# ElevenLabs API (for high-quality TTS)
# Get your API key from: https://elevenlabs.io/
ELEVENLABS_API_KEY=your_elevenlabs_api_key_here

# VseGPT API (alternative LLM provider)
# Get your API key from your VseGPT provider
VSEGPT_API_KEY=your_vsegpt_api_key_here

# Google Cloud (for Speech-to-Text and other services)
# Path to your service account JSON file
GOOGLE_APPLICATION_CREDENTIALS=/data/credentials/google-cloud-service-account.json
# Your Google Cloud Project ID
GOOGLE_CLOUD_PROJECT_ID=your_google_project_id

# ============================================================
# COMPONENT CONFIGURATION
# ============================================================

# Enable/disable core components.
# NOTE: components are normally set in your config.toml (e.g. config.toml), NOT here.
# Env vars OVERRIDE the config file, so these are left COMMENTED — `cp docs/env-example.txt .env`
# must not force a partial component set. (Older copies crashed at startup: they enabled TTS but
# used the WRONG field `AUDIO_OUTPUT`, leaving Audio off → "TTS component requires Audio component".)
# Field names must match the schema (AUDIO, not AUDIO_OUTPUT); if you enable TTS, also enable AUDIO.
# LOCVEIL_VOICE_COMPONENTS__TTS=true
# LOCVEIL_VOICE_COMPONENTS__AUDIO=true          # required whenever TTS is enabled
# LOCVEIL_VOICE_COMPONENTS__ASR=true
# LOCVEIL_VOICE_COMPONENTS__LLM=true
# LOCVEIL_VOICE_COMPONENTS__NLU=true
# LOCVEIL_VOICE_COMPONENTS__TEXT_PROCESSOR=true
# LOCVEIL_VOICE_COMPONENTS__INTENT_SYSTEM=true
# Web API is a SYSTEM capability, not a component:
# LOCVEIL_VOICE_SYSTEM__WEB_API_ENABLED=true
